ESP32

Ένα καλώδιο USB Type C χρησιμοποιείται για τη σύνδεση του ESP32 με τη συσκευή android. Η συσκευή υποστηρίζει σύνδεση Wifi
που συνδέεται στο δίκτυο κινητής τηλεφωνίας. Χρησιμοποιείται C++ για Arduino για τον προγραμματισμό του ESP32.
Το ESP32 λαμβάνει εντολές και ελέγχει το τιμόνι (Servo) & το Electronic Speed Controller (ESC).


#include <WiFi.h>
#include <PubSubClient.h>
#include <ESP32Servo.h>
#include <ArduinoJson.h>

// το hotspot του σπιτιού ή οποιοδήποτε Wi-Fi με σύνδεση Internet
const char* SSID     = "Redmi10s";
const char* PASS     = "12345678";

// δημόσιος broker
const char* MQTT_BROKER = "broker.hivemq.com";
const int   MQTT_PORT   = 1883;
const char* TOPIC_INPUT = "myrc/car1/input";

WiFiClient    net;
PubSubClient  mqtt(net);

// pins
const int STEERING_PIN = 18;
const int ESC_PIN      = 19;

// servos
Servo steeringServo, esc;

// state
int steeringAngle = 90;
int throttlePWM   = 0;

// parse incoming MQTT JSON
void onMqttMessage(char* topic, byte* payload, unsigned int len) {
    // copy into a String
    String msg;
    for(int i = 0; i < len; i++) msg += (char)payload[i];

    // expected: {"angle":…, "pwm":…, "gear":…}
    StaticJsonDocument<200> doc;
    DeserializationError err = deserializeJson(doc, msg);
    if(err) {
        Serial.println("❌ Αποτυχία ανάλυσης JSON");
        return;
    }
    steeringAngle = doc["angle"];
    throttlePWM   = doc["pwm"];
    Serial.printf("📥 MQTT → angle=%d pwm=%d\n", steeringAngle, throttlePWM);
}

void setup() {
    Serial.begin(115200);
    WiFi.begin(SSID, PASS);
    while(WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
    }
    Serial.println("\n✅ Wi-Fi ενεργό, IP=" + WiFi.localIP().toString());

    mqtt.setServer(MQTT_BROKER, MQTT_PORT);
    mqtt.setCallback(onMqttMessage);
    while(!mqtt.connected()) {
        Serial.print("⏳ Σύνδεση με MQTT…");
        if(mqtt.connect("ESP32RCCar")) {
            Serial.println("✅");
        } else {
            Serial.print(" αποτυχία, rc=");
            Serial.print(mqtt.state());
            Serial.println(" επανάληψη σε 2s");
            delay(2000);
        }
    }
    mqtt.subscribe(TOPIC_INPUT);
    Serial.println("🔴 Εγγραφή στο " + String(TOPIC_INPUT));

    // attach servos/ESC
    steeringServo.setPeriodHertz(50);
    esc.setPeriodHertz(50);
    steeringServo.attach(STEERING_PIN, 500, 2500);
    esc.attach(ESC_PIN, 1000, 2000);

    // arm ESC
    esc.writeMicroseconds(2000); delay(500);
    esc.writeMicroseconds(1000); delay(500);
    esc.writeMicroseconds(1500); delay(500);
}

void loop() {
    mqtt.loop();  // pump incoming MQTT

    // drive
    steeringServo.write(steeringAngle);
    int us = map(throttlePWM, -255, 255, 1000, 2000);
    esc.writeMicroseconds(constrain(us, 1000, 2000));
}

Αρχικοποίηση Ελέγχου Κινητήρα

Αυτή η εφαρμογή Python Flask παρέχει διεπαφή ελέγχου σε πραγματικό χρόνο για το τηλεκατευθυνόμενο αυτοκίνητο χρησιμοποιώντας χειριστήριο PS4 🎮 και Android κινητό ως κάμερα και server.


import os
import time
import json
import eventlet
eventlet.monkey_patch()

from flask import render_template

from flask import Flask, render_template, Response
from flask_socketio import SocketIO
import paho.mqtt.client as mqtt
import cv2    # ← νέο

# — Ρύθμιση Flask / Socket.IO —
app = Flask(__name__)
app.config['SECRET_KEY'] = 'rc-car-secret'
socketio = SocketIO(
        app,
        async_mode='eventlet',
        cors_allowed_origins="*",
        logger=True,
        engineio_logger=True
)

@app.route('/phone')
def phone():
        # εμφανίζει το πρότυπο του broadcaster κινητού
        return render_template('phone_loc.html')

# — Ρύθμιση MQTT (χωρίς αλλαγές) —
MQTT_BROKER = os.getenv('MQTT_BROKER', 'broker.hivemq.com')
MQTT_PORT   = int(os.getenv('MQTT_PORT', '1883'))
TOPIC_INPUT = 'myrc/car1/input'
mqttc = mqtt.Client()
mqttc.connect(MQTT_BROKER, MQTT_PORT, keepalive=60)
mqttc.loop_start()

# — Gear & smoothing (χωρίς αλλαγές) —
GEAR_SCALE      = [0.0, 0.10, 0.3, 0.5, 0.7, 1.0]
SMOOTH          = 0.2
steer_smooth    = 0.0
throttle_smooth = 0.0
scale_smooth    = GEAR_SCALE[1]

# — Ρύθμιση λήψης κάμερας —  
# 0 = προεπιλεγμένη webcam. Αλλάξτε σε RTSP/MJPEG URL για IP cams ή "http://:8080/video"
camera = cv2.VideoCapture("http://100.83.193.15:8080/video") 
if not camera.isOpened():
        raise RuntimeError("Δεν ήταν δυνατή η εκκίνηση της κάμερας.")

def gen_frames():
        """Παράγει καρέ κάμερας ως multipart MJPEG."""
        while True:
                success, frame = camera.read()
                if not success:
                        break
                # προαιρετικά: resize, flip, overlay κλπ εδώ
                ret, buffer = cv2.imencode('.jpg', frame)
                jpg = buffer.tobytes()
                yield (b'--frame\r\n'
                             b'Content-Type: image/jpeg\r\n\r\n' + jpg + b'\r\n')

@app.route('/video_feed')
def video_feed():
        return Response(
                gen_frames(),
                mimetype='multipart/x-mixed-replace; boundary=frame'
        )

@app.route('/')
def index():
        # index.html θα ενσωματώσει το /video_feed
        return render_template('index.html')


@socketio.on('connect')
def handle_connect(auth):
        print(f"[{time.strftime('%H:%M:%S')}] Συνδέθηκε το dashboard")
        socketio.emit('telemetry', {
                'steering': 90,
                'throttle': 0,
                'gear': 1,
                'speed': 0.0
        })
@socketio.on('location')
def handle_location(data):
        # data == {'lat':…, 'lng':…}
        print(f"[{time.strftime('%H:%M:%S')}] 📍 λήψη τοποθεσίας:", data)
        socketio.emit('location', data)


@socketio.on('controller_input')
def handle_input(data):
        global steer_smooth, throttle_smooth, scale_smooth

        # (η υπάρχουσα εξομάλυνση, MQTT publish, telemetry emit…)

        raw_steer = data.get('steering', 0.0)
        raw_thr   = data.get('throttle', 0.0)
        raw_brk   = data.get('brake', 0.0)
        gear      = data.get('gear', 1)

        steer_smooth += SMOOTH * (raw_steer - steer_smooth)
        angle = int((steer_smooth * 0.5 + 0.5) * 180)

        forward = -raw_brk if raw_brk > 0.1 else raw_thr
        throttle_smooth += SMOOTH * (forward - throttle_smooth)

        target_scale = GEAR_SCALE[gear]
        scale_smooth += SMOOTH * (target_scale - scale_smooth)

        pwm = int(throttle_smooth * scale_smooth * 255)

        payload = {'angle': angle,'pwm': pwm,'gear': gear}
        mqttc.publish(TOPIC_INPUT, json.dumps(payload), qos=0)

        speed = abs(pwm) * 0.2 * gear
        socketio.emit('telemetry', {
                'steering': angle,
                'throttle': pwm,
                'gear': gear,
                'speed': round(speed,1)
        })
@socketio.on('restart_esp')
def restart_esp():
        # ο κώδικας σας για επανεκκίνηση ESP (π.χ. εντολή serial ή MQTT)
        print("Ζητήθηκε επανεκκίνηση ESP")

@socketio.on('restart_hotspot')
def restart_hotspot():
        # ο κώδικας σας για επανεκκίνηση hotspot Android
        print("Ζητήθηκε επανεκκίνηση hotspot")

if __name__ == '__main__':
        print("Έναρξη server στο http://0.0.0.0:5000/")
        socketio.run(app, host='0.0.0.0', port=5000, debug=True)
